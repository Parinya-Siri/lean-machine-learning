      rw he at this,

      exact (mem_union _ _ _).2 (or.inr this) },

    { apply PG.comp he'.2 he.2 e'.open_target e.open_target,
      apply e'.continuous_inv_fun.preimage_open_of_open e'.open_target e.open_target }
  end,
